$('.ui-draglist-inputrow').on('keyup', function(e) {
  if (e.keyCode == 13 &&
    $(this).children('.ui-draglist-attrinput').val() &&
    $(this).children('.ui-draglist-syninput').val()) {

    var row = $('<li />', { class: 'ui-draglist-row' });
    $('<p />', {
      class: 'ui-draglist-attr',
      text: $(this).children('.ui-draglist-attrinput').val()
    }).appendTo(row);
    $('<p />', {
      class: 'ui-draglist-syn',
      text: $(this).children('.ui-draglist-syninput').val()
    }).appendTo(row);
    $('<i />', { class: 'ui-draglist-sort' }).appendTo(row);
    $('<i />', { class: 'ui-draglist-del' }).appendTo(row);
    $('.ui-draglist').append(row);

    $(this).children('.ui-draglist-attrinput').val('');
    $(this).children('.ui-draglist-syninput').val('');
  }
});
